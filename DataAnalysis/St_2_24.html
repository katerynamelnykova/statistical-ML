Have total emissions from PM2.5 decreased in the United States from 1999 to 2008? Make a plot showing the total PM2.5 emission from all sources for each of the years 1999, 2002, 2005, and 2008.

total_emissions = df_summary.groupby("year")["Emissions"].sum().reset_index()

def plot_emissions_bars(data, title=''): 
    plt.figure(figsize=(8, 5))
    ax = sns.barplot(data=data, x="year", y="Emissions", palette="mako")

    for p in ax.containers:
        ax.bar_label(p, fmt="%.0f", label_type="edge", fontsize=10, padding=3)

    x_positions = [bar.get_x() + bar.get_width() / 2 for bar in ax.patches[:len(data)]]

    plt.plot(x_positions, data["Emissions"], marker="o", color="pink", linestyle="-", linewidth=2)
    plt.title(title)
    plt.ylabel("Total Emissions (tons)")
    plt.xlabel("Year")
    plt.grid(axis="y", linestyle="--", alpha=0.7)
    plt.show()

plot_emissions_bars(total_emissions, "Total PM2.5 Emissions in the US (1999–2008)")

combined_vehicle_data = pd.concat([baltimore_vehicle_emissions, la_vehicle_emissions])

def plot_emissions_two_cities(data, title=''):
    plt.figure(figsize=(8, 5))
    ax = sns.barplot(data=data, x="year", y="Emissions", hue="City", palette="mako")

    for p in ax.containers:
        ax.bar_label(p, fmt="%.0f", label_type="edge", fontsize=10, padding=3)

    cities = data["City"].unique()
    colors = {"Baltimore": "pink", "Los Angeles": "lightblue"}
    
    for city in cities:
        city_data = data[data["City"] == city].sort_values("year")
        years = city_data["year"].values
        emissions = city_data["Emissions"].values
        
        x_positions = list(range(len(years)))

        plt.plot(x_positions, emissions, marker="o", linestyle="-", linewidth=2, color=colors[city], label=f"{city}")

    plt.xticks(ticks=range(len(years)), labels=years)
    plt.title(title)
    plt.ylabel("Total Emissions (tons)")
    plt.xlabel("Year")
    plt.legend(title="City")
    plt.grid(axis="y", linestyle="--", alpha=0.7)
    plt.show()


plot_emissions_two_cities(combined_vehicle_data, "Vehicle PM2.5 Emissions: Baltimore vs. Los Angeles (1999-2008)")

By analyzing the graph above, we can see that the number of emissions in LA significantly exceeds the number of emissions in Baltimore. In both cities, we can see a decrease in emissions between 1999 and 2008, but in percentage terms, Baltimore's emissions change more significantly (by 65.8%) compared to the decrease in emissions in LA (by 35.4%).
